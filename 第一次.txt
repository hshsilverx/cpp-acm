1.4算可用的插板插孔
#include <iostream>
 
using namespace std;
 
int num,n,m,i,j,counts;
int main()
{
    m=0;
    cin>>num;
    for (i=0;i<num;i++)
    {
        cin>>n;
        for(j=0;j<n;j++)
        {
            cin>>m;
            counts+=m;
        }
        cout<<counts-(n-1)<<"\n";
        counts=0;
    }
}
 
1.5计算获胜次数
#include <iostream>
 
using namespace std;
 
int i,n,m,mary,john;
 
int main()
{
    while(1)
    {
        cin>>n;
        if(n!=0)
        {
            mary=john=0;
            for (i=0;i<n;i++)
            {
                cin>>m;
                if(m==0)
                {
                    mary++;
                }else
                {
                    john++;
                }
            }
            cout<<"Mary won "<<mary<<" times and John won "<<john<<" times"<<endl;
        }
        else
        {
            break;
        }
    }
}
